			\item Biconjugate Gradient (BICG),
			\item Quasi-Minimal Residual (QMR),
			\item Conjugate Gradient Squared (CGS),
			\item Biconjugate Gradient Stabilized (BiCGSTAB),
			\item Chebyshev Iteration.

		\end{itemize}
