\begin{frame}
  \begin{itemize}
    \item wielomian stopnia $N-1$,
    \item współczynniki: $C(1), C(2), \dots, C(N)$; $C(N)$ -- przy $x^{N-1}$
  \end{itemize}

  $$\begin{array}{ll}
  P = C(N) \cdot x + C(N-1) & \rightarrow P(x) \\
  P_1 = C(N) & \rightarrow P'(x)
  \end{array}$$
\end{frame}
